28int main(
29 int argc,
30 char *argv[]) {
31
32 static iasi_rad_t *iasi_rad;
33
34 static wave_t wave;
35
36 static FILE *out;
37
38 static double mu, nesr, sigma;
39
40 static int ichan, itrack, ix, iy;
41
42 /* Check arguments... */
43 if (argc < 4)
44 ERRMSG("Give parameters: <ctl> <iasi_l1_file> <noise.tab>");
45
46 /* Allocate... */
47 ALLOC(iasi_rad, iasi_rad_t, 1);
48
49 /* Read IASI data... */
50 printf("Read IASI data: %s\n", argv[2]);
51 iasi_read(argv[2], iasi_rad);
52
53 /* Create file... */
54 printf("Write noise data: %s\n", argv[3]);
55 if (!(out = fopen(argv[3], "w")))
56 ERRMSG("Cannot create file!");
57
58 /* Write header... */
59 fprintf(out,
60 "# $1 = track index\n"
61 "# $2 = channel index\n"
62 "# $3 = wavenumber [1/cm]\n"
63 "# $4 = mean BT [K]\n"
64 "# $5 = NEDT [K]\n" "# $6 = NESR [W/(m^2 sr cm^-1)]\n");
65
66 /* Analyze blocks of data... */
67 for (itrack = 0; itrack < iasi_rad->ntrack; itrack += 60) {
68
69 /* Write empty line... */
70 fprintf(out, "\n");
71
72 /* Loop over channels... */
73 for (ichan = 0; ichan < IASI_L1_NCHAN; ichan++) {
74
75 /* Set wave struct... */
76 wave.nx = L1_NXTRACK;
77 wave.ny = 0;
78 for (iy = itrack; iy < GSL_MIN(itrack + 60, iasi_rad->ntrack); iy++) {
79 for (ix = 0; ix < wave.nx; ix++)
80 wave.temp[ix][wave.ny] = BRIGHT(iasi_rad->Rad[iy][ix][ichan],
81 iasi_rad->freq[ichan]);
82 wave.ny++;
83 }
84
85 /* Check number of data points... */
86 if (wave.ny >= 55) {
87
88 /* Get noise... */
89 noise(&wave, &mu, &sigma);
90
91 /* Get NESR... */
92 nesr = PLANCK(mu + sigma, iasi_rad->freq[ichan])
93 - PLANCK(mu, iasi_rad->freq[ichan]);
94
95 /* Write output... */
96 if (gsl_finite(sigma))
97 fprintf(out, "%d %d %.4f %g %g %g\n", itrack, ichan,
98 iasi_rad->freq[ichan], mu, sigma, nesr);
99 }
100 }
101 }
102
103 /* Close file... */
104 fclose(out);
105
106 /* Free... */
107 free(iasi_rad);
108
109 return EXIT_SUCCESS;
110}
