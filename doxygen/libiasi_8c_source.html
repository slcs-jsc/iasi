57void background_poly_help(
58 double *xx,
59 double *yy,
60 int n,
61 int dim) {
62
63 gsl_multifit_linear_workspace *work;
64 gsl_matrix *cov, *X;
65 gsl_vector *c, *x, *y;
66
67 double chisq, xx2[WX > WY ? WX : WY], yy2[WX > WY ? WX : WY];
68
69 size_t i, i2, n2 = 0;
70
71 /* Check for nan... */
72 for (i = 0; i < (size_t) n; i++)
73 if (gsl_finite(yy[i])) {
74 xx2[n2] = xx[i];
75 yy2[n2] = yy[i];
76 n2++;
77 }
78 if ((int) n2 < dim || n2 < 0.9 * n) {
79 for (i = 0; i < (size_t) n; i++)
80 yy[i] = GSL_NAN;
81 return;
82 }
83
84 /* Allocate... */
85 work = gsl_multifit_linear_alloc((size_t) n2, (size_t) dim);
86 cov = gsl_matrix_alloc((size_t) dim, (size_t) dim);
87 X = gsl_matrix_alloc((size_t) n2, (size_t) dim);
88 c = gsl_vector_alloc((size_t) dim);
89 x = gsl_vector_alloc((size_t) n2);
90 y = gsl_vector_alloc((size_t) n2);
91
92 /* Compute polynomial fit... */
93 for (i = 0; i < (size_t) n2; i++) {
94 gsl_vector_set(x, i, xx2[i]);
95 gsl_vector_set(y, i, yy2[i]);
96 for (i2 = 0; i2 < (size_t) dim; i2++)
97 gsl_matrix_set(X, i, i2, pow(gsl_vector_get(x, i), (double) i2));
98 }
99 gsl_multifit_linear(X, y, c, cov, &chisq, work);
100 for (i = 0; i < (size_t) n; i++)
101 yy[i] = gsl_poly_eval(c->data, (int) dim, xx[i]);
102
103 /* Free... */
104 gsl_multifit_linear_free(work);
105 gsl_matrix_free(cov);
106 gsl_matrix_free(X);
107 gsl_vector_free(c);
108 gsl_vector_free(x);
109 gsl_vector_free(y);
110}

114void background_poly(
115 wave_t *wave,
116 int dim_x,
117 int dim_y) {
118
119 double help[WX], x[WX], x2[WY], y[WX], y2[WY];
120
121 int ix, iy;
122
123 /* Copy temperatures to background... */
124 for (ix = 0; ix < wave->nx; ix++)
125 for (iy = 0; iy < wave->ny; iy++) {
126 wave->bg[ix][iy] = wave->temp[ix][iy];
127 wave->pt[ix][iy] = 0;
128 }
129
130 /* Check parameters... */
131 if (dim_x <= 0 && dim_y <= 0)
132 return;
133
134 /* Compute fit in x-direction... */
135 if (dim_x > 0)
136 for (iy = 0; iy < wave->ny; iy++) {
137 for (ix = 0; ix <= 53; ix++) {
138 x[ix] = (double) ix;
139 y[ix] = wave->bg[ix][iy];
140 }
141 background_poly_help(x, y, 54, dim_x);
142 for (ix = 0; ix <= 29; ix++)
143 help[ix] = y[ix];
144
145 for (ix = 6; ix <= 59; ix++) {
146 x[ix - 6] = (double) ix;
147 y[ix - 6] = wave->bg[ix][iy];
148 }
149 background_poly_help(x, y, 54, dim_x);
150 for (ix = 30; ix <= 59; ix++)
151 help[ix] = y[ix - 6];
152
153 for (ix = 0; ix < wave->nx; ix++)
154 wave->bg[ix][iy] = help[ix];
155 }
156
157 /* Compute fit in y-direction... */
158 if (dim_y > 0)
159 for (ix = 0; ix < wave->nx; ix++) {
160 for (iy = 0; iy < wave->ny; iy++) {
161 x2[iy] = (int) iy;
162 y2[iy] = wave->bg[ix][iy];
163 }
164 background_poly_help(x2, y2, wave->ny, dim_y);
165 for (iy = 0; iy < wave->ny; iy++)
166 wave->bg[ix][iy] = y2[iy];
167 }
168
169 /* Recompute perturbations... */
170 for (ix = 0; ix < wave->nx; ix++)
171 for (iy = 0; iy < wave->ny; iy++)
172 wave->pt[ix][iy] = wave->temp[ix][iy] - wave->bg[ix][iy];
173}

177void background_smooth(
178 wave_t *wave,
179 int npts_x,
180 int npts_y) {
181
182 const double dmax = 2500.0;
183
184 static double help[WX][WY];
185
186 /* Check parameters... */
187 if (npts_x <= 0 && npts_y <= 0)
188 return;
189
190 /* Smooth background... */
191 for (int ix = 0; ix < wave->nx; ix++)
192 for (int iy = 0; iy < wave->ny; iy++) {
193
194 /* Init... */
195 int n = 0;
196 help[ix][iy] = 0;
197
198 /* Set maximum range... */
199 const int dx = GSL_MIN(GSL_MIN(npts_x, ix), wave->nx - 1 - ix);
200 const int dy = GSL_MIN(GSL_MIN(npts_y, iy), wave->ny - 1 - iy);
201
202 /* Average... */
203 for (int i = ix - dx; i <= ix + dx; i++)
204 for (int j = iy - dy; j <= iy + dy; j++)
205 if (fabs(wave->x[ix] - wave->x[i]) < dmax &&
206 fabs(wave->y[iy] - wave->y[j]) < dmax) {
207 help[ix][iy] += wave->bg[i][j];
208 n++;
209 }
210
211 /* Normalize... */
212 if (n > 0)
213 help[ix][iy] /= n;
214 else
215 help[ix][iy] = GSL_NAN;
216 }
217
218 /* Recalculate perturbations... */
219 for (int ix = 0; ix < wave->nx; ix++)
220 for (int iy = 0; iy < wave->ny; iy++) {
221 wave->bg[ix][iy] = help[ix][iy];
222 wave->pt[ix][iy] = wave->temp[ix][iy] - wave->bg[ix][iy];
223 }
224}

228void gauss(
229 wave_t *wave,
230 double fwhm) {
231
232 static double help[WX][WY];
233
234 /* Check parameters... */
235 if (fwhm <= 0)
236 return;
237
238 /* Compute sigma^2... */
239 const double sigma2 = gsl_pow_2(fwhm / 2.3548);
240
241 /* Loop over data points... */
242 for (int ix = 0; ix < wave->nx; ix++)
243 for (int iy = 0; iy < wave->ny; iy++) {
244
245 /* Init... */
246 double wsum = 0;
247 help[ix][iy] = 0;
248
249 /* Average... */
250 for (int ix2 = 0; ix2 < wave->nx; ix2++)
251 for (int iy2 = 0; iy2 < wave->ny; iy2++) {
252 const double d2 = gsl_pow_2(wave->x[ix] - wave->x[ix2])
253 + gsl_pow_2(wave->y[iy] - wave->y[iy2]);
254 if (d2 <= 9 * sigma2) {
255 const double w = exp(-d2 / (2 * sigma2));
256 wsum += w;
257 help[ix][iy] += w * wave->pt[ix2][iy2];
258 }
259 }
260
261 /* Normalize... */
262 wave->pt[ix][iy] = help[ix][iy] / wsum;
263 }
264}

268void hamming(
269 wave_t *wave,
270 int niter) {
271
272 static double help[WX][WY];
273
274 /* Iterations... */
275 for (int iter = 0; iter < niter; iter++) {
276
277 /* Filter in x direction... */
278 for (int ix = 0; ix < wave->nx; ix++)
279 for (int iy = 0; iy < wave->ny; iy++)
280 help[ix][iy]
281 = 0.23 * wave->pt[ix > 0 ? ix - 1 : ix][iy]
282 + 0.54 * wave->pt[ix][iy]
283 + 0.23 * wave->pt[ix < wave->nx - 1 ? ix + 1 : ix][iy];
284
285 /* Filter in y direction... */
286 for (int ix = 0; ix < wave->nx; ix++)
287 for (int iy = 0; iy < wave->ny; iy++)
288 wave->pt[ix][iy]
289 = 0.23 * help[ix][iy > 0 ? iy - 1 : iy]
290 + 0.54 * help[ix][iy]
291 + 0.23 * help[ix][iy < wave->ny - 1 ? iy + 1 : iy];
292 }
293}

297void iasi_read(
298 char *filename,
299 iasi_rad_t *iasi_rad) {
300
301 const char *product_class;
302
303 coda_product *pf;
304
305 coda_cursor cursor;
306
307 iasi_raw_t *iasi_raw;
308
309 int i, j, w, tr1, tr2, tr1_lpm, tr1_rpm, tr2_lpm, tr2_rpm,
310 ichan, mdr_i, num_dims = 1;
311
312 long dim[] = { 1, 2, 3, 4, 5, 6, 7, 8, 9 };
313
314 short int IDefScaleSondNbScale, IDefScaleSondNsfirst[10],
315 IDefScaleSondNslast[10], IDefScaleSondScaleFactor[10];
316
317 float sc, scaling[IASI_L1_NCHAN];
318
319 /* Initialize CODA... */
320 coda_init();
321
322 /* Allocate... */
323 ALLOC(iasi_raw, iasi_raw_t, 1);
324
325 /* Open IASI file... */
326 CODA(coda_open(filename, &pf));
327 CODA(coda_get_product_class(pf, &product_class));
328 CODA(coda_cursor_set_product(&cursor, pf));
329
330 /* Get scaling parameters... */
331 CODA(coda_cursor_goto_record_field_by_name(&cursor, "GIADR_ScaleFactors"));
332
333 CODA(coda_cursor_goto_record_field_by_name
334 (&cursor, "IDefScaleSondNbScale"));
335 CODA(coda_cursor_read_int16(&cursor, &IDefScaleSondNbScale));
336 CODA(coda_cursor_goto_parent(&cursor));
337
338 CODA(coda_cursor_goto_record_field_by_name
339 (&cursor, "IDefScaleSondNsfirst"));
340 CODA(coda_cursor_read_int16_array
341 (&cursor, IDefScaleSondNsfirst, coda_array_ordering_c));
342 CODA(coda_cursor_goto_parent(&cursor));
343
344 CODA(coda_cursor_goto_record_field_by_name(&cursor, "IDefScaleSondNslast"));
345 CODA(coda_cursor_read_int16_array
346 (&cursor, IDefScaleSondNslast, coda_array_ordering_c));
347 CODA(coda_cursor_goto_parent(&cursor));
348
349 CODA(coda_cursor_goto_record_field_by_name
350 (&cursor, "IDefScaleSondScaleFactor"));
351 CODA(coda_cursor_read_int16_array
352 (&cursor, IDefScaleSondScaleFactor, coda_array_ordering_c));
353
354 /* Compute scaling factors... */
355 for (ichan = 0; ichan < IASI_L1_NCHAN; ichan++)
356 scaling[ichan] = GSL_NAN;
357 for (i = 0; i < IDefScaleSondNbScale; i++) {
358 sc = (float) pow(10.0, -IDefScaleSondScaleFactor[i]);
359 for (ichan = IDefScaleSondNsfirst[i] - 1;
360 ichan < IDefScaleSondNslast[i]; ichan++) {
361 w = ichan - IASI_IDefNsfirst1b + 1;
362 if (w >= 0 && w < IASI_L1_NCHAN)
363 scaling[w] = sc;
364 }
365 }
366
367 /* Get number of tracks in record... */
368 CODA(coda_cursor_goto_root(&cursor));
369 CODA(coda_cursor_goto_record_field_by_name(&cursor, "MDR"));
370 CODA(coda_cursor_get_array_dim(&cursor, &num_dims, dim));
371 iasi_raw->ntrack = dim[0];
372
373 /* Read tracks one by one... */
374 for (mdr_i = 0; mdr_i < iasi_raw->ntrack; mdr_i++) {
375
376 /* Reset cursor position... */
377 CODA(coda_cursor_goto_root(&cursor));
378
379 /* Move cursor to radiation data... */
380 CODA(coda_cursor_goto_record_field_by_name(&cursor, "MDR"));
381 CODA(coda_cursor_goto_array_element_by_index(&cursor, mdr_i));
382 CODA(coda_cursor_goto_record_field_by_name(&cursor, "MDR"));
383 CODA(coda_cursor_goto_record_field_by_name(&cursor, "GS1cSpect"));
384 CODA(coda_cursor_read_int16_array
385 (&cursor, &iasi_raw->Radiation[mdr_i][0][0][0],
386 coda_array_ordering_c));
387
388 /* Read time... */
389 CODA(coda_cursor_goto_parent(&cursor));
390 CODA(coda_cursor_goto_record_field_by_name(&cursor, "OnboardUTC"));
391 CODA(coda_cursor_read_double_array
392 (&cursor, &iasi_raw->Time[mdr_i][0], coda_array_ordering_c));
393
394 /* Read coordinates... */
395 CODA(coda_cursor_goto_parent(&cursor));
396 CODA(coda_cursor_goto_record_field_by_name(&cursor, "GGeoSondLoc"));
397 CODA(coda_cursor_read_double_array
398 (&cursor, &iasi_raw->Loc[mdr_i][0][0][0], coda_array_ordering_c));
399
400 /* Read satellite altitude... */
401 CODA(coda_cursor_goto_parent(&cursor));
402 CODA(coda_cursor_goto_record_field_by_name(&cursor,
403 "EARTH_SATELLITE_DISTANCE"));
404 CODA(coda_cursor_read_uint32(&cursor, &iasi_raw->Sat_z[mdr_i]));
405
406 /* Read spectral range... */
407 iasi_raw->IDefSpectDWn1b[mdr_i] = IASI_IDefSpectDWn1b / 100.0;
408
409 CODA(coda_cursor_goto_parent(&cursor));
410 CODA(coda_cursor_goto_record_field_by_name(&cursor, "IDefNsfirst1b"));
411 CODA(coda_cursor_read_int32(&cursor, &iasi_raw->IDefNsfirst1b[mdr_i]));
412 if (iasi_raw->IDefNsfirst1b[mdr_i] != IASI_IDefNsfirst1b)
413 ERRMSG("Unexpected value for IDefNsfirst1b!");
414
415 CODA(coda_cursor_goto_parent(&cursor));
416 CODA(coda_cursor_goto_record_field_by_name(&cursor, "IDefNslast1b"));
417 CODA(coda_cursor_read_int32(&cursor, &iasi_raw->IDefNslast1b[mdr_i]));
418 if (iasi_raw->IDefNslast1b[mdr_i] != IASI_IDefNslast1b)
419 ERRMSG("Unexpected value for IDefNslast1b!");
420
421 /* Compute wavenumber... */
422 if (mdr_i == 0)
423 for (i = 0; i < IASI_L1_NCHAN; i++)
424 iasi_raw->Wavenumber[i] =
425 iasi_raw->IDefSpectDWn1b[mdr_i] *
426 (float) (iasi_raw->IDefNsfirst1b[mdr_i] + i - 1);
427 }
428
429 /* Close file... */
430 CODA(coda_close(pf));
431
432 /* Finalize CODA... */
433 coda_done();
434
435 /* Set number of tracks... */
436 iasi_rad->ntrack = (int) (iasi_raw->ntrack * 2);
437
438 /* Copy wavenumbers... */
439 for (ichan = 0; ichan < IASI_L1_NCHAN; ichan++)
440 iasi_rad->freq[ichan] = iasi_raw->Wavenumber[ichan];
441
442 /* Copy footprint data... */
443 for (mdr_i = 0; mdr_i < iasi_raw->ntrack; mdr_i++) {
444 tr1 = mdr_i * 2;
445 tr2 = mdr_i * 2 + 1;
446 tr1_lpm = 3;
447 tr1_rpm = 0;
448 tr2_lpm = 2;
449 tr2_rpm = 1;
450
451 /* Copy time (2x2 matrix has same measurement time)... */
452 for (i = 0; i < IASI_NXTRACK; i++) {
453 iasi_rad->Time[tr1][i * 2] = iasi_raw->Time[mdr_i][i];
454 iasi_rad->Time[tr1][i * 2 + 1] = iasi_raw->Time[mdr_i][i];
455 iasi_rad->Time[tr2][i * 2] = iasi_raw->Time[mdr_i][i];
456 iasi_rad->Time[tr2][i * 2 + 1] = iasi_raw->Time[mdr_i][i];
457 }
458
459 /* Copy location... */
460 for (i = 0; i < IASI_NXTRACK; i++) {
461 iasi_rad->Longitude[tr1][i * 2] = iasi_raw->Loc[mdr_i][i][tr1_lpm][0];
462 iasi_rad->Longitude[tr1][i * 2 + 1] =
463 iasi_raw->Loc[mdr_i][i][tr1_rpm][0];
464 iasi_rad->Latitude[tr1][i * 2] = iasi_raw->Loc[mdr_i][i][tr1_lpm][1];
465 iasi_rad->Latitude[tr1][i * 2 + 1] =
466 iasi_raw->Loc[mdr_i][i][tr1_rpm][1];
467
468 iasi_rad->Longitude[tr2][i * 2] = iasi_raw->Loc[mdr_i][i][tr2_lpm][0];
469 iasi_rad->Longitude[tr2][i * 2 + 1] =
470 iasi_raw->Loc[mdr_i][i][tr2_rpm][0];
471 iasi_rad->Latitude[tr2][i * 2] = iasi_raw->Loc[mdr_i][i][tr2_lpm][1];
472 iasi_rad->Latitude[tr2][i * 2 + 1] =
473 iasi_raw->Loc[mdr_i][i][tr2_rpm][1];
474 }
475
476 /* Copy satellite location (we only have one height value)... */
477 iasi_rad->Sat_lon[tr1] = iasi_rad->Longitude[tr1][28];
478 iasi_rad->Sat_lat[tr1] = iasi_rad->Latitude[tr1][28];
479 iasi_rad->Sat_lon[tr2] = iasi_rad->Longitude[tr2][28];
480 iasi_rad->Sat_lat[tr2] = iasi_rad->Latitude[tr2][28];
481 iasi_rad->Sat_z[tr1] =
482 iasi_raw->Sat_z[mdr_i] / 1000.0 - wgs84(iasi_rad->Sat_lat[tr1]);
483 iasi_rad->Sat_z[tr2] =
484 iasi_raw->Sat_z[mdr_i] / 1000.0 - wgs84(iasi_rad->Sat_lat[tr2]);
485
486 /* Copy radiation data... */
487 for (i = 0; i < IASI_NXTRACK; i++) {
488 for (ichan = 0; ichan < IASI_L1_NCHAN; ichan++) {
489 sc = scaling[ichan] * 100.0f;
490 iasi_rad->Rad[tr1][i * 2][ichan] =
491 iasi_raw->Radiation[mdr_i][i][tr1_lpm][ichan] * sc;
492 iasi_rad->Rad[tr1][i * 2 + 1][ichan] =
493 iasi_raw->Radiation[mdr_i][i][tr1_rpm][ichan] * sc;
494 iasi_rad->Rad[tr2][i * 2][ichan] =
495 iasi_raw->Radiation[mdr_i][i][tr2_lpm][ichan] * sc;
496 iasi_rad->Rad[tr2][i * 2 + 1][ichan] =
497 iasi_raw->Radiation[mdr_i][i][tr2_rpm][ichan] * sc;
498 }
499 }
500 }
501
502 /* Check radiance data... */
503 for (i = 0; i < iasi_rad->ntrack; i++)
504 for (j = 0; j < L1_NXTRACK; j++)
505 if (iasi_rad->Rad[i][j][6753] > iasi_rad->Rad[i][j][6757]
506 || iasi_rad->Rad[i][j][6753] < 0)
507 for (ichan = 0; ichan < IASI_L1_NCHAN; ichan++)
508 iasi_rad->Rad[i][j][ichan] = GSL_NAN;
509
510 /* Free... */
511 free(iasi_raw);
512}

516void median(
517 wave_t *wave,
518 int dx) {
519
520 static double data[WX * WY], help[WX][WY];
521
522 /* Check parameters... */
523 if (dx <= 0)
524 return;
525
526 /* Loop over data points... */
527 for (int ix = 0; ix < wave->nx; ix++)
528 for (int iy = 0; iy < wave->ny; iy++) {
529
530 /* Init... */
531 size_t n = 0;
532
533 /* Get data... */
534 for (int ix2 = GSL_MAX(ix - dx, 0);
535 ix2 < GSL_MIN(ix + dx, wave->nx - 1); ix2++)
536 for (int iy2 = GSL_MAX(iy - dx, 0);
537 iy2 < GSL_MIN(iy + dx, wave->ny - 1); iy2++) {
538 data[n] = wave->pt[ix2][iy2];
539 n++;
540 }
541
542 /* Normalize... */
543 gsl_sort(data, 1, n);
544 help[ix][iy] = gsl_stats_median_from_sorted_data(data, 1, n);
545 }
546
547 /* Loop over data points... */
548 for (int ix = 0; ix < wave->nx; ix++)
549 for (int iy = 0; iy < wave->ny; iy++)
550 wave->pt[ix][iy] = help[ix][iy];
551}

553/*****************************************************************************/
554
555void noise(
556 wave_t *wave,
557 double *mu,
558 double *sig) {
559
560 int ix, ix2, iy, iy2, n = 0, okay;
561
562 /* Init... */
563 *mu = 0;
564 *sig = 0;
565
566 /* Estimate noise (Immerkaer, 1996)... */
567 for (ix = 1; ix < wave->nx - 1; ix++)
568 for (iy = 1; iy < wave->ny - 1; iy++) {
569
570 /* Check data... */
571 okay = 1;
572 for (ix2 = ix - 1; ix2 <= ix + 1; ix2++)
573 for (iy2 = iy - 1; iy2 <= iy + 1; iy2++)
574 if (!gsl_finite(wave->temp[ix2][iy2]))
575 okay = 0;
576 if (!okay)
577 continue;
578
579 /* Get mean noise... */
580 n++;
581 *mu += wave->temp[ix][iy];
582 *sig += gsl_pow_2(+4. / 6. * wave->temp[ix][iy]
583 - 2. / 6. * (wave->temp[ix - 1][iy]
584 + wave->temp[ix + 1][iy]
585 + wave->temp[ix][iy - 1]
586 + wave->temp[ix][iy + 1])
587 + 1. / 6. * (wave->temp[ix - 1][iy - 1]
588 + wave->temp[ix + 1][iy - 1]
589 + wave->temp[ix - 1][iy + 1]
590 + wave->temp[ix + 1][iy + 1]));
591 }
592
593 /* Normalize... */
594 *mu /= (double) n;
595 *sig = sqrt(*sig / (double) n);
596}

600void pert2wave(
601 pert_t *pert,
602 wave_t *wave,
603 int track0,
604 int track1,
605 int xtrack0,
606 int xtrack1) {
607
608 double x0[3], x1[3];
609
610 int itrack, ixtrack;
611
612 /* Check ranges... */
613 track0 = GSL_MIN(GSL_MAX(track0, 0), pert->ntrack - 1);
614 track1 = GSL_MIN(GSL_MAX(track1, 0), pert->ntrack - 1);
615 xtrack0 = GSL_MIN(GSL_MAX(xtrack0, 0), pert->nxtrack - 1);
616 xtrack1 = GSL_MIN(GSL_MAX(xtrack1, 0), pert->nxtrack - 1);
617
618 /* Set size... */
619 wave->nx = xtrack1 - xtrack0 + 1;
620 if (wave->nx > WX)
621 ERRMSG("Too many across-track values!");
622 wave->ny = track1 - track0 + 1;
623 if (wave->ny > WY)
624 ERRMSG("Too many along-track values!");
625
626 /* Loop over footprints... */
627 for (itrack = track0; itrack <= track1; itrack++)
628 for (ixtrack = xtrack0; ixtrack <= xtrack1; ixtrack++) {
629
630 /* Get distances... */
631 if (itrack == track0) {
632 wave->x[0] = 0;
633 if (ixtrack > xtrack0) {
634 geo2cart(0, pert->lon[itrack][ixtrack - 1],
635 pert->lat[itrack][ixtrack - 1], x0);
636 geo2cart(0, pert->lon[itrack][ixtrack],
637 pert->lat[itrack][ixtrack], x1);
638 wave->x[ixtrack - xtrack0] =
639 wave->x[ixtrack - xtrack0 - 1] + DIST(x0, x1);
640 }
641 }
642 if (ixtrack == xtrack0) {
643 wave->y[0] = 0;
644 if (itrack > track0) {
645 geo2cart(0, pert->lon[itrack - 1][ixtrack],
646 pert->lat[itrack - 1][ixtrack], x0);
647 geo2cart(0, pert->lon[itrack][ixtrack],
648 pert->lat[itrack][ixtrack], x1);
649 wave->y[itrack - track0] =
650 wave->y[itrack - track0 - 1] + DIST(x0, x1);
651 }
652 }
653
654 /* Save geolocation... */
655 wave->time = pert->time[(track0 + track1) / 2][(xtrack0 + xtrack1) / 2];
656 wave->z = 0;
657 wave->lon[ixtrack - xtrack0][itrack - track0] =
658 pert->lon[itrack][ixtrack];
659 wave->lat[ixtrack - xtrack0][itrack - track0] =
660 pert->lat[itrack][ixtrack];
661
662 /* Save temperature data... */
663 wave->temp[ixtrack - xtrack0][itrack - track0]
664 = pert->bt[itrack][ixtrack];
665 wave->bg[ixtrack - xtrack0][itrack - track0]
666 = pert->bt[itrack][ixtrack] - pert->pt[itrack][ixtrack];
667 wave->pt[ixtrack - xtrack0][itrack - track0]
668 = pert->pt[itrack][ixtrack];
669 wave->var[ixtrack - xtrack0][itrack - track0]
670 = pert->var[itrack][ixtrack];
671 }
672}

756/*****************************************************************************/
757
758void variance(
759 wave_t *wave,
760 double dh) {
761
762 double dh2, mu, help;
763
764 int dx, dy, ix, ix2, iy, iy2, n;
765
766 /* Check parameters... */
767 if (dh <= 0)
768 return;
769
770 /* Compute squared radius... */
771 dh2 = gsl_pow_2(dh);
772
773 /* Get sampling distances... */
774 dx =
775 (int) (dh / fabs(wave->x[wave->nx - 1] - wave->x[0]) * (wave->nx - 1.0) +
776 1);
777 dy =
778 (int) (dh / fabs(wave->y[wave->ny - 1] - wave->y[0]) * (wave->ny - 1.0) +
779 1);
780
781 /* Loop over data points... */
782 for (ix = 0; ix < wave->nx; ix++)
783 for (iy = 0; iy < wave->ny; iy++) {
784
785 /* Init... */
786 mu = help = 0;
787 n = 0;
788
789 /* Get data... */
790 for (ix2 = GSL_MAX(ix - dx, 0); ix2 <= GSL_MIN(ix + dx, wave->nx - 1);
791 ix2++)
792 for (iy2 = GSL_MAX(iy - dy, 0); iy2 <= GSL_MIN(iy + dy, wave->ny - 1);
793 iy2++)
794 if ((gsl_pow_2(wave->x[ix] - wave->x[ix2])
795 + gsl_pow_2(wave->y[iy] - wave->y[iy2])) <= dh2)
796 if (gsl_finite(wave->pt[ix2][iy2])) {
797 mu += wave->pt[ix2][iy2];
798 help += gsl_pow_2(wave->pt[ix2][iy2]);
799 n++;
800 }
801
802 /* Compute local variance... */
803 if (n > 1)
804 wave->var[ix][iy] = help / n - gsl_pow_2(mu / n);
805 else
806 wave->var[ix][iy] = GSL_NAN;
807 }
808}

811
812double wgs84(
813 double lat) {
814
815 const double a = 6378.1370, b = 6356.7523;
816
817 double cphi, sphi;
818
819 cphi = cos(lat * M_PI / 180.);
820 sphi = sin(lat * M_PI / 180.);
821
822 return sqrt((gsl_pow_2(a * a * cphi) + gsl_pow_2(b * b * sphi))
823 / (gsl_pow_2(a * cphi) + gsl_pow_2(b * sphi)));
824}
